def attach(attachable)
  blob_was = blob if attached?
  blob = create_blob_from(attachable)

  unless blob == blob_was
    transaction do
      detach
      write_attachment build_attachment(blob: blob)
    end

    blob_was.purge_later if blob_was && dependent == :purge_later
  end
end
